EasyBbk  1.0.0
"Easy BBK" is a trial to provide a standardized, visualized and user friendly access to bio-brick system by simplify and interassociate the searching, comparing, designing and uploading of bio-bricks. These four main functions constitute the "Easy BBK" client-side which is realized by pure java and supported by a remote server through the internet. We hope that you will have better experiences when exploring the biobrick world using "Easy BBK".

206  private void exeOperation(SketchOperation operation)
207  {
208  switch (operation.operationType)
209  { case SketchOperation.ADD:

211  componentList.add(compToAdd);
212  if (compToAdd.primaryType.equals(BackBone.class.getSimpleName()))
213  backBoneList.add(compToAdd.toBackBone());
214  break;
215  case SketchOperation.REMOVE:
216  Component compToRemove = (Component) operation.previous;
217  componentList.remove(compToRemove);
218  if (compToRemove.primaryType.equals(BackBone.class.getSimpleName()))
219  backBoneList.remove(compToRemove.toBackBone());
220  break;
221  case SketchOperation.MODIFY:
222  Component component = findComponentByID(operation.ID);
223  Object object = operation.following;
224  switch (operation.attributeType)
225  { case SketchOperation.TYPE_STRING:
226  component.setString((String) object); break;
227  case SketchOperation.TYPE_BOUNDS:
228  component.setBounds((Rectangle) object);
229  if (isMoveEvent(operation.previous, operation.following))
230  if (component.primaryType.equals(BackBone.class.getSimpleName()))
231  onLinkage(component.toBackBone(), operation.previous, operation.following);
232  else if (component.primaryType.equals(BackBone.class.getSimpleName()))
233  onLinkage(component.toRelation(), operation.previous, operation.following);
234  break;
235  case SketchOperation.TYPE_THICKNESS:
236  component.setThickness((Double) object); break;
237  case SketchOperation.TYPE_FONT:
238  component.setFont((Font) object); break;
239  case SketchOperation.TYPE_COLOR:
240  component.setColor((Color) object); break;
