510  center = new Point(Integer.parseInt(xy[0]), Integer.parseInt(xy[1]));
511  }

513  // about curve
514  Element eleCurve = (Element) componentNode.getElementsByTagName("curve").item(0);
515  String curveStr = eleCurve.getFirstChild().getNodeValue();
516  ArrayList<Point> curve = null;
517  if ( !curveStr.equals("null") )
518  { curve = new ArrayList<Point>();
519  String[] pointStrList = curveStr.split(",");
520  for (String pointStr : pointStrList)
521  if ( !pointStr.equals("") ) // there WILL be an "" in each list
522  { String[] xy = pointStr.split(" ");
523  curve.add(new Point(Integer.parseInt(xy[0]), Integer.parseInt(xy[1])));
524  }

527  // about size
528  Element eleSize = (Element) componentNode.getElementsByTagName("size").item(0);
529  String sizeStr = eleSize.getFirstChild().getNodeValue();
530  Double size = null;
531  if ( !sizeStr.equals("null") )
532  size = Double.parseDouble(sizeStr);
533 
534  // about font
535  Element eleFont = (Element) componentNode.getElementsByTagName("font").item(0);
536  String fontStr = eleFont.getFirstChild().getNodeValue();
537  Font font = null;
538  if ( !fontStr.equals("null") )
539  { String[] tokens = fontStr.split(",");
540  font = new Font
541  (tokens[0], Integer.parseInt(tokens[1]), Integer.parseInt(tokens[2]));
542  }
543 
544  // about color
545  Element eleColor = (Element) componentNode.getElementsByTagName("color").item(0);
546  String colorStr = eleColor.getFirstChild().getNodeValue();
547  Color color = null;
548  if ( !colorStr.equals("null") )
549  color = new Color(Integer.parseInt(colorStr));
550 
551  // about children
552  Element eleChildren = (Element) componentNode.getElementsByTagName("children").item(0);
553  String childrenStr = eleChildren.getFirstChild().getNodeValue();
554  ArrayList<Integer> children = null;
555  if ( !childrenStr.equals("null") )
556  { children = new ArrayList<Integer>();
557  String[] childStrList = childrenStr.split(" ");
558  for (String childStr : childStrList)
559  if ( !childStr.equals("") ) // there WILL be an "" in each list
560  children.add(Integer.parseInt(childStr));
561  }
562 
563  Component component = null;
564  if (primaryType.equals(Label.class.getSimpleName()))
565  component = new Label(ID, string, center, font, color);
566  else if (primaryType.equals(BioBrick.class.getSimpleName()))
567  component = new BioBrick(ID, string, secondaryType, center, color);
568  else if (primaryType.equals(Protein.class.getSimpleName()))
569  component = new Protein(ID, secondaryType, center, color);
570  else if (primaryType.equals(BackBone.class.getSimpleName()))
571  component = new BackBone(ID, center, (int)size.doubleValue(), children);
572  else if (primaryType.equals(Relation.class.getSimpleName()))
573  component = new Relation(ID, secondaryType, curve, color, (int)size.doubleValue());
574  else if (primaryType.equals(BioVector.class.getSimpleName()))
575  component = new BioVector(ID, secondaryType, center, size);
576 
577  return component;
578  }
